#########

####APP1
server {
  listen 80;
  server_name rubyshop.ga www.rubyshop.com;

  location / {
    proxy_pass http://localhost:3001;
    proxy_set_header Host $http_host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
  }
}
